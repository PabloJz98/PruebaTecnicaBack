CREATE TABLE EMPLOYESS(
	IDS INTEGER PRIMARY KEY,
	GENDER_ID INTEGER,
	JOB_ID INTEGER,
	NAMESS VARCHAR(255),
	LAST_NAME VARCHAR(255),
	BIRTHDATE DATE,
		FOREIGN KEY(GENDER_ID) 
       	REFERENCES GENDERS(IDS),
		FOREIGN KEY(JOB_ID) 
       	REFERENCES JOBS(IDS)
)


CREATE TABLE EMPLOYEE_WORKED_HOURS(
	IDS INTEGER PRIMARY KEY,
	EMPLOYEE_ID INTEGER,
	WORKED_HOURS INTEGER,
	WORKED_DATE DATE,
		FOREIGN KEY(EMPLOYEE_ID) 
       	REFERENCES EMPLOYESS(IDS)
)

CREATE TABLE GENDERS(
	IDS INTEGER PRIMARY KEY,
	NAMESS VARCHAR(255)
)

CREATE TABLE JOBS(
	IDS INTEGER PRIMARY KEY,
	NAMESS VARCHAR(255),
	SALARY FLOAT
)
